```
using System;
using System.IO;
using System.Collections;

class MainClass {
    public static void Main (string[] args) {
        //------------------
        // 輸入檔
        //------------------
        StreamReader file = new StreamReader("data.csv"); 
        
        
        //----------------------------
        // 依序讀入各行資料
        //----------------------------
        string line;
        
        while((line = file.ReadLine()) != null)  
        {  
            // 將讀入整行以逗號分割  
            string[] items = line.Split(',');
            
            // 取出各項目資料
            string no = items[0];
            string gender = items[1];
            int chi = Int32.Parse(items[2]);
            int eng = Int32.Parse(items[3]);
            int mat = Int32.Parse(items[4]);
            int soc = Int32.Parse(items[5]);
            int nat = Int32.Parse(items[6]);
            int lec = Int32.Parse(items[7]);
            
            // 進行某些處理
            int tot = chi + eng + mat + soc + nat + lec;
            Console.WriteLine(tot);
        }  

        //----------------------------
        // 關檔
        //----------------------------       
        file.Close();  
    }
}
```
