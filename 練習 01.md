### 2.MainClass.cs

```
using System;
using System.IO;

class MainClass {
    public static void Main (string[] args) {
        //------------------
        // 輸入檔
        //------------------
        StreamReader file = new StreamReader("data.csv");         
        
        //----------------------------
        // 依序讀入各行資料
        //----------------------------
        string line;
        
        while((line = file.ReadLine()) != null)  
        {  
            // 將讀入整行以逗號分割  
            string[] items = line.Split(',');
            
            // 取出各項目資料
            // 序號(0), 性別(1), 入學方式(2), 戶籍(3)
            // 國文(4), 英文(5), 數學(6), 社會(7), 自然(8), 作文(9)
            int no = Int32.Parse(items[0]);
            string gender = items[1];
            string entrance = items[2];
            string city = items[3];
            int chi = Int32.Parse(items[4]);
            int eng = Int32.Parse(items[5]);
            int mat = Int32.Parse(items[6]);
            int soc = Int32.Parse(items[7]);
            int nat = Int32.Parse(items[8]);
            int lec = Int32.Parse(items[9]);
            
            // 進行某些處理
            int tot = chi + eng + mat + soc + nat + lec;
            Console.WriteLine("no:{0}  total:{1}", no, tot);
        }  

        //----------------------------
        // 關檔
        //----------------------------       
        file.Close();  
    }
}
```
